# Note: Cascaded foreign key actions do not activate triggers
/*
[CONSTRAINT [symbol]] FOREIGN KEY
    [index_name] (col_name, ...)
    REFERENCES tbl_name (col_name,...)
    [ON DELETE reference_option]
    [ON UPDATE reference_option]

reference_option:
    RESTRICT | CASCADE | SET NULL | NO ACTION | SET DEFAULT
*/
# Single-column foreign key
CREATE TABLE alpha
(
    id INT NOT NULL,
    PRIMARY KEY (id)
);
CREATE TABLE beta
(
    id INT,
    alpha_id INT,
    INDEX alp_ind (alpha_id),
    FOREIGN KEY (alpha_id)
    REFERENCES alpha(id)
    ON DELETE CASCADE
);

#  Foreign keys, one foreign key references a two-column index and the other references a single-column index:
CREATE TABLE apache
(
    type INT NOT NULL,
    id INT NOT NULL,
    stock DECIMAL,
    PRIMARY KEY(type, id)
);
CREATE TABLE friend
(
    id INT NOT NULL,
    PRIMARY KEY (id)
);
CREATE TABLE apache_order
(
    no INT NOT NULL AUTO_INCREMENT,
    apache_type INT NOT NULL,
    apache_id INT NOT NULL,
    friend_id INT NOT NULL,
    PRIMARY KEY(no),
    INDEX (apache_type, apache_id),
    INDEX (friend_id),
    FOREIGN KEY (apache_type, apache_id)
    REFERENCES apache(type, id)
    ON UPDATE CASCADE
    ON DELETE RESTRICT
);

# Adding Foreign Key Constraints
# The foreign key can be self referential (referring to the same table)
/*
ALTER TABLE tbl_name
    ADD [CONSTRAINT [symbol]] FOREIGN KEY
    [index_name] (col_name, ...)
    REFERENCES tbl_name (col_name,...)
    [ON DELETE reference_option]
    [ON UPDATE reference_option]
*/
DELETE FROM `sample`.`apache_order`;
